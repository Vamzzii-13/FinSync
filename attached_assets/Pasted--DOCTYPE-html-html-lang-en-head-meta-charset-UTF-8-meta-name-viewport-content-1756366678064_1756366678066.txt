    <script>
        // --- Page Navigation Logic ---
        const loginPage = document.getElementById('loginPage');
        const dashboardPage = document.getElementById('dashboardPage');
        const returnsPage = document.getElementById('returnsPage');

        const loginBtn = document.getElementById('loginBtn');
        const headerLoginBtn = document.getElementById('headerLoginBtn');
        const searchBtn = document.getElementById('searchBtn');

        function showPage(pageToShow) {
            // Hide all pages first
            loginPage.classList.add('hidden');
            dashboardPage.classList.add('hidden');
            returnsPage.classList.add('hidden');

            // Show the requested page
            pageToShow.classList.remove('hidden');
            window.scrollTo(0, 0); // Scroll to top on page change
        }

        // Event listener for the main login button
        loginBtn.addEventListener('click', () => {
            // In a real app, you would validate credentials here
            showPage(dashboardPage);
        });

        // Event listener for the header login button (to go back to login page)
        headerLoginBtn.addEventListener('click', () => {
            showPage(loginPage);
        });

        // Event listener for the search button on the dashboard
        searchBtn.addEventListener('click', () => {
            showPage(returnsPage);
        });

        // --- Upload and Status Change Logic ---
        const uploadButtons = document.querySelectorAll('.upload-btn');

        uploadButtons.forEach(button => {
            button.addEventListener('click', (event) => {
                // Create a file input element
                const fileInput = document.createElement('input');
                fileInput.type = 'file';
                fileInput.style.display = 'none';

                // Listen for file selection
                fileInput.addEventListener('change', () => {
                    if (fileInput.files.length > 0) {
                        // Find the parent card of the button that was clicked
                        const card = event.target.closest('.card');
                        if (card) {
                            // Find the status text element within that card
                            const statusText = card.querySelector('.status-text');
                            if (statusText) {
                                // Update status to 'Filed'
                                statusText.textContent = 'Filed';
                                statusText.classList.remove('status-not-filed');
                                statusText.classList.add('status-filed');
                            }
                        }
                    }
                    // Clean up the file input element
                    document.body.removeChild(fileInput);
                });

                // Append and trigger the file input
                document.body.appendChild(fileInput);
                fileInput.click();
            });
        });


        // Initially show the login page
        document.addEventListener('DOMContentLoaded', () => {
            showPage(loginPage);
        });
    </script>
